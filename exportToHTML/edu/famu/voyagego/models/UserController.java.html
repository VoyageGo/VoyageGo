package edu.famu.voyagego.models;

import edu.famu.voyagego.models.User;
import com.google.cloud.firestore.Firestore;
import com.google.firebase.cloud.FirestoreClient;
import org.springframework.web.bind.annotation.*;

import java.util.concurrent.ExecutionException;

@RestController
@RequestMapping("/users")
public class UserController {

    private final Firestore db;

    public UserController() {
        this.db = FirestoreClient.getFirestore(); // Use Firestore singleton
    }

    @PostMapping("/signup")
    public String createUser(@RequestBody User user) {
        try {
            db.collection("users").document(user.getUserId()).set(user).get();
            return "User created successfully!";
        } catch (Exception e) {
            e.printStackTrace();
            return "Error: " + e.getMessage();
        }
    }

    @GetMapping("/{userId}")
    public User getUser(@PathVariable String userId) throws ExecutionException, InterruptedException {
        return db.collection("users").document(userId).get().get().toObject(User.class);
    }

    @PutMapping("/{userId}")
    public String updateUser(@PathVariable String userId, @RequestBody User user) {
        try {
            db.collection("users").document(userId).set(user).get();
            return "User updated successfully!";
        } catch (Exception e) {
            return "Error: " + e.getMessage();
        }
    }

    @DeleteMapping("/{userId}")
    public String deleteUser(@PathVariable String userId) {
        try {
            db.collection("users").document(userId).delete().get();
            return "User deleted successfully!";
        } catch (Exception e) {
            return "Error: " + e.getMessage();
        }
    }
}
