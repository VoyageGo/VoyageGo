import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/api/rewards")
public class RewardsController {
    @GetMapping("/balance/{userId}")
    public int getRewardPoints(@PathVariable String userId) {
        // Fetch the user's current reward points balance
        return 1000;
    }

    @GetMapping("/available")
    public List<String> getAvailableRewards() {
        return List.of("Reward 1", "Reward 2");
    }

    @PostMapping("/redeem")
    public String redeemReward(@RequestBody String rewardId) {
        // Redeem a reward
        return "Reward redeemed successfully!";
    }

    @GetMapping("/history/{userId}")
    public List<String> getRewardHistory(@PathVariable String userId) {

    }
}
